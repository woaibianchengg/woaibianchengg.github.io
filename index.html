饿了么数据接口

发表于 2018-04-16

在最新版本的vue-cli 没有’dev-server.js’以及’dev-client.js’两个文件之后，如果想对饿了吗数据进行获取，应该这样去操作！
在新版本的build文件夹内的webpack.dev.conf.js文件夹内，进行相关的数据获取。

//首先
const express = require(‘express’)
const app = express()
var appData = require(‘../data.json’)
var seller = appData.seller
var goods = appData.goods
var ratings = appData.ratings

//添加路由
var apiRoutes = express.Router()
app.use(‘/api’, apiRoutes)

然后找到devServer
在对象里添加
before(app) {
app.get(‘/api/seller’, (req, res) => {
//你的Json内容
res.json({
errno: 0,
data: seller
})
}),
}
